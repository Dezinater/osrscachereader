IndexType.valueOf = (id) => {
    var values = Object.values(IndexType);
    var keys = Object.keys(IndexType);
    for (var i = 0; i < values.length; i++) {
        if (id == values[i].id)
            return IndexType[keys[i]];
    }
    return undefined;
}

Object.freeze(IndexType);

export default IndexType;

ConfigType.valueOf = (id) => {
	var values = Object.values(ConfigType);
	var keys = Object.keys(ConfigType);
	for (var i = 0; i < values.length; i++) {
		if (id == values[i].id)
			return ConfigType[keys[i]];
	}
	return undefined;
}

Object.freeze(ConfigType);

export default ConfigType;
